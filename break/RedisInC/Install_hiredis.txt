# hiredis in C
# ref; https://developer.redis.com/develop/C/#step-5-test-the-code


# Install the pre-requisites; C compiler and Make file
sudo apt install gcc make

# Install and compile hiredis
wget https://github.com/redis/hiredis/archive/master.zip
chmod +x master.zip
unzip master.zip
mv hiredis-master hiredis && cd hiredis
make
sudo make install

# Compile the code (source code in ref)
gcc redistest.c -o redistest -I /usr/local/include/hiredis -lhiredis


# Ex. Output; PING "Hello World"
RESPONSE: Hello World
